# Ensure 'Created Date' and 'Closed Date' are in datetime format
df['Created Date'] = pd.to_datetime(df['Created Date'])
df['Closed Date'] = pd.to_datetime(df['Closed Date'])

# Calculate the time difference between 'Closed Date' and 'Created Date' in days
df['Request_Closing_Time_Days'] = (df['Closed Date'] - df['Created Date']).dt.days

# Select relevant columns and remove rows with missing values
avg_response = df[['Complaint Type', 'Request_Closing_Time_Days']].dropna()

# Group the data by 'Complaint Type'
grouped = avg_response.groupby('Complaint Type')

# Create a list of groups where each group contains the 'Request_Closing_Time_Days' values for each complaint type
groups = [group['Request_Closing_Time_Days'].values for name, group in grouped if len(group) > 1]

# Perform the ANOVA test to check if there's a significant difference between the means of the groups
from scipy.stats import f_oneway
f_stat, p_val = f_oneway(*groups)

# Print the F-statistic and p-value, rounded to an appropriate number of decimal places
print("F-statistic:", round(f_stat, 4))
print("p-value:", round(p_val, 4))

# Check if the p-value is less than the significance level (0.05), and make a decision about the null hypothesis
if p_val < 0.05:
    print("Reject the null hypothesis: The average response time differs across complaint types.")
else:
    print("Accept the null hypothesis: No significant difference in response time among complaint types.")
